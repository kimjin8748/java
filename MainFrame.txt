		private void showFood(String foodName) {
		    // 데이터베이스 연결 정보
		    String url = "jdbc:mysql://localhost/cban";
		    String username = "root";
		    String password = "1234";

		    try {
		        // JDBC 드라이버 로드
		        Class.forName("com.mysql.cj.jdbc.Driver");

		        // 데이터베이스 연결
		        Connection connection = DriverManager.getConnection(url, username, password);

		        // 쿼리 실행
		        Statement statement = connection.createStatement();

		        // foodName을 사용하여 동적으로 쿼리 생성
		        String query = "SELECT * FROM newtable WHERE menu = '" + foodName + "'";
		        ResultSet resultSet = statement.executeQuery(query);

		        try {
		            while (resultSet.next()) {
		                String menu = resultSet.getString("menu"); // 가져올 열 이름 지정
		                String price = resultSet.getString("price");
		                String number = resultSet.getString("number");
		                String totalprice = resultSet.getString("totalprice");
		                listModel.addElement(menu);
		                listModel1.addElement(price);
		                listModel2.addElement(number);
		                listModel3.addElement(totalprice);		                
		            }
		        } catch (SQLException e1) {
		            e1.printStackTrace();
		        }

		        // 자원 해제
		        resultSet.close();
		        statement.close();
		        connection.close();
		    } catch (Exception e) {
		        e.printStackTrace();
		    }
		
	}

	private static void addPopup(Component component, final JPopupMenu popup) {
		component.addMouseListener(new MouseAdapter() {
			public void mousePressed(MouseEvent e) {
				if (e.isPopupTrigger()) {
					showMenu(e);
				}
			}
			public void mouseReleased(MouseEvent e) {
				if (e.isPopupTrigger()) {
					showMenu(e);
				}
			}
			private void showMenu(MouseEvent e) {
				popup.show(e.getComponent(), e.getX(), e.getY());
			}
		});
	}
